## Grid: API

```tsx
import { Grid } from '@niceltd/sol-react';
```

### Component API Description

#### Props

<table>
  <tr>
    <th>Name</th>
    <th>Type</th>
    <th>Description</th>
  </tr>
  <tr>
    <td>dataSource</td>
    <td>IDatasource | ICompatibleDatasource | IServerSideDatasource | ICompatibleServerSideDatasource | undefined</td>
    <td>Defines the data source for the ag-grid. The types are specific to ag-grid.</td>
  </tr>
  <tr>
    <td>maxAmountOfPages</td>
    <td>number</td>
    <td>Specifies the maximum number of pages that the grid can display.</td>
  </tr>
  <tr>
    <td>rangeOfDisplayedItems</td>
    <td>string</td>
    <td>Specifies the range of items to be displayed in the pager. This is a string value.</td>
  </tr>
  <tr>
    <td>columnDefs</td>
    <td>(ColDef | ColGroupDef)[] | null | undefined</td>
    <td>Defines the column definitions for the ag-grid.</td>
  </tr>
  <tr>
    <td>defaultColDef</td>
    <td>ColDef</td>
    <td>Defines the default column definitions for the ag-grid.</td>
  </tr>
  <tr>
    <td>rowData</td>
    <td>any[]</td>
    <td>Specifies the data to be displayed in the grid.</td>
  </tr>
  <tr>
    <td>gridOptions</td>
    <td>GridOptionsCellSpinnerProp</td>
    <td>Defines the options for the ag-grid including row selection, sorting, and filtering. For checkbox-selection columns, sol-grid applies a default accessible checkbox cell renderer when no custom `cellRenderer` is provided.</td>
  </tr>
  <tr>
    <td>pagination</td>
    <td>boolean</td>
    <td>Enables or disables pagination in the grid. If true, a pager will be displayed.</td>
  </tr>
  <tr>
    <td>maxItemsPerPage</td>
    <td>number</td>
    <td>Sets the maximum number of items to be displayed per page when pagination is enabled.</td>
  </tr>
  <tr>
    <td>gridState</td>
    <td>unknown</td>
    <td>Sets the initial state of the grid including column visibility and order.</td>
  </tr>
  <tr>
    <td>gridSort</td>
    <td>ColumnState[] | any</td>
    <td>Sets the initial sort order of the grid.</td>
  </tr>
  <tr>
    <td>frameworkComponents</td>
    <td>any</td>
    <td>Register custom cell renderer, editor, or header components for use in column definitions.</td>
  </tr>
  <tr>
    <td>pageNumber</td>
    <td>number</td>
    <td>Controls the current page programmatically (1-based). Defaults to 0 (no navigation).</td>
  </tr>
  <tr>
    <td>selectRowEnabled</td>
    <td>boolean</td>
    <td>Enables visual single-row selection highlighting on click. Defaults to true.</td>
  </tr>
  <tr>
    <td>ariaLabel</td>
    <td>string</td>
    <td>Accessible name for the grid when no external label element is provided.</td>
  </tr>
  <tr>
    <td>ariaLabelledby</td>
    <td>string | null</td>
    <td>ID of an external element that labels the grid.</td>
  </tr>
  <tr>
    <td>i18n</td>
    <td>GridI18n</td>
    <td>Optional i18n overrides for all grid labels and announcements.</td>
  </tr>
</table>
